List domains script

This script shows how to list and extract global information from your domains.
 1# If you want to run this script, move it one folder above (next to cf.py)
 2
 3import os, dotenv
 4from datetime import datetime
 5
 6from cf_rules import Cloudflare
 7
 8dotenv.load_dotenv(".env")
 9
10cf = Cloudflare("my_expressions")
11cf.auth(os.environ.get("EMAIL"), os.environ.get("KEY"))
12
13# Get domains
14
15domains = cf.get_domains()
16
17print(f"There are {domains['count']} domains available for your account:")
18
19print(" ".join(domains["domains"]))
20
21# Same as
22
23domains = cf.domains
24
25print(f"There are {len(domains)} domains available for your account:")
26
27print(" ".join([x.name for x in domains]))
28
29
30
31for domain in domains:
32    print(f"{domain['name']} is {domain['status']}")
33    created = datetime.fromisoformat(domain['created_on'].replace("Z", ""))
34    print("Created on " + created.strftime("%Y-%m-%d %H:%M:%S"))
